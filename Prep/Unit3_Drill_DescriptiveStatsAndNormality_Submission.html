import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
%matplotlib inline

def plot_hist(distribution):
    #Plot a histogram.
    plt.hist(distribution)

    # Add a vertical line at the mean.
    plt.axvline(distribution.mean(), color='b', linestyle='solid', linewidth=2)

    # Add a vertical line at one standard deviation above the mean.
    plt.axvline(distribution.mean() + distribution.std(), color='b', linestyle='dashed', linewidth=2)

    # Add a vertical line at one standard deviation below the mean.
    plt.axvline(distribution.mean() - distribution.std(), color='b', linestyle='dashed', linewidth=2) 

    # Print the histogram
    print('Mean: ', distribution.mean())
    print('Std: ', distribution.std())
    plt.show()

  • The mean is almost 15 (10 + 5), so the sum of both means of normal1 and normal2
  • The standard error is not the sum of both origin standard errors, but always higher than 1. So the standard error is not linear additive
  • nothing more special, just a normal distribution
